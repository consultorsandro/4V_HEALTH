1/// This module provides functionality to calculate and classify Body Fat Percentage (PGC)
2/// using the Deurenberg formula, with classification by sex and by sex+age.
3/// It follows SOLID principles for maintainability and extensibility.
4
5use crate::bmi::calculator::{BmiCalculator, BmiData};
6use crate::metabolism::tmb::Gender;
7
8/// Data structure for Body Fat calculation input.
9pub struct BodyFatData {
10    pub weight: f32,
11    pub height: f32,
12    pub age: u32,
13    pub gender: Gender,
14}
15
16/// Enum for general body fat classification by sex.
17pub enum BodyFatSexCategory {
18    Essential,
19    Athlete,
20    Fitness,
21    Acceptable,
22    Obesity,
23}
24
25/// Enum for body fat classification by sex and age.
26pub enum BodyFatAgeCategory {
27    Low,
28    Normal,
29    High,
30    VeryHigh,
31}
32
33/// Main calculator for Body Fat Percentage.
34pub struct BodyFatCalculator;
35
36impl BodyFatCalculator {
37    /// Calculates BMI using the existing BMI module.
38    pub fn calculate_bmi(data: &BodyFatData) -> f32 {
39        let bmi_data = BmiData { weight: data.weight, height: data.height };
40        BmiCalculator::calculate(&bmi_data)
41    }
42
43    /// Calculates Body Fat Percentage using the Deurenberg formula.
44    /// sex: 1 for male, 0 for female
45    pub fn calculate_pgc(bmi: f32, age: u32, gender: &Gender) -> f32 {
46        let sex = match gender {
47            Gender::Male => 1.0,
48            Gender::Female => 0.0,
49        };
50        (1.20 * bmi) + (0.23 * age as f32) - (10.8 * sex) - 5.4
51    }
52
53    /// Classifies Body Fat Percentage by sex.
54    pub fn classify_by_sex(pgc: f32, gender: &Gender) -> BodyFatSexCategory {
55        match gender {
56            Gender::Male => {
57                match pgc {
58                    x if x < 6.0 => BodyFatSexCategory::Essential,
59                    x if x < 14.0 => BodyFatSexCategory::Athlete,
60                    x if x < 18.0 => BodyFatSexCategory::Fitness,
61                    x if x < 25.0 => BodyFatSexCategory::Acceptable,
62                    _ => BodyFatSexCategory::Obesity,
63                }
64            }
65            Gender::Female => {
66                match pgc {
67                    x if x < 14.0 => BodyFatSexCategory::Essential,
68                    x if x < 21.0 => BodyFatSexCategory::Athlete,
69                    x if x < 25.0 => BodyFatSexCategory::Fitness,
70                    x if x < 32.0 => BodyFatSexCategory::Acceptable,
71                    _ => BodyFatSexCategory::Obesity,
72                }
73            }
74        }
75    }
76
77    /// Classifies Body Fat Percentage by sex and age.
78    pub fn classify_by_age(pgc: f32, age: u32, gender: &Gender) -> BodyFatAgeCategory {
79        let (low, normal, high, very_high) = match gender {
80            Gender::Male => match age {
81                20..=29 => (7.0, 19.0, 24.0, 25.0),
82                30..=39 => (8.0, 20.0, 25.0, 26.0),
83                40..=49 => (10.0, 22.0, 27.0, 28.0),
84                50..=59 => (11.0, 23.0, 28.0, 29.0),
85                _ => (13.0, 25.0, 30.0, 31.0), // 60+
86            },
87            Gender::Female => match age {
88                20..=29 => (16.0, 27.0, 32.0, 33.0),
89                30..=39 => (17.0, 28.0, 33.0, 34.0),
90                40..=49 => (18.0, 29.0, 34.0, 35.0),
91                50..=59 => (19.0, 30.0, 35.0, 36.0),
92                _ => (20.0, 31.0, 36.0, 37.0), // 60+
93            },
94        };
95
96        if pgc < low {
97            BodyFatAgeCategory::Low
98        } else if pgc <= normal {
99            BodyFatAgeCategory::Normal
100        } else if pgc <= high {
101            BodyFatAgeCategory::High
102        } else {
103            BodyFatAgeCategory::VeryHigh
104        }
105    }
106
107    /// Returns a string with the result for the user.
108    pub fn evaluation_result(
109        pgc: f32,
110        gender: &Gender,
111        age: u32,
112        sex_category: &BodyFatSexCategory,
113        age_category: &BodyFatAgeCategory,
114    ) -> String {
115        let sex_str = match gender {
116            Gender::Male => "male",
117            Gender::Female => "female",
118        };
119        let sex_class = match sex_category {
120            BodyFatSexCategory::Essential => "Essential to life",
121            BodyFatSexCategory::Athlete => "Athlete",
122            BodyFatSexCategory::Fitness => "Fitness",
123            BodyFatSexCategory::Acceptable => "Acceptable",
124            BodyFatSexCategory::Obesity => "Obesity",
125        };
126        let age_class = match age_category {
127            BodyFatAgeCategory::Low => "Low",
128            BodyFatAgeCategory::Normal => "Normal",
129            BodyFatAgeCategory::High => "High",
130            BodyFatAgeCategory::VeryHigh => "Very High",
131        };
132        format!(
133            "Your Body Fat Percentage (PGC), sex [{}] is: {}\n\
134             Your PGC for sex [{}] and age group [{}] (OMC Standard) is: {}",
135            sex_str, sex_class, sex_str, age, age_class
136        )
137    }
